In JS all objects are “bags” for propertires according to the official documentation.

All objects link to a prototype object.

This link from a prototype to the next protytype goes on until desired attribute is on.


Attribute

function Person(first, last, age, eye) {
this.firstName = first;
this.lastName = last;
this.age = age;
this.eyeColor = eye;
}
var child=new Person("John", "Doe", 50, "blue");
console.log(child.age);//50
Person.prototype.major="informationsystem";//add a new attribute the the child object
console.log(child.major);//informationsystem

The child object inherit from the person object with its own attributes

Also, you can add a new attribute to the person object if you want, like I add a "major" attribute above

Function

function Person(first, last, age, eye) {
this.firstName = first;
this.lastName = last;
this.age = age;
this.eyeColor = eye;
this.getAge=function(){
return age;
}
}
var child=new Person("John", "Doe", 50, "blue");
console.log(child.getAge());
Person.prototype.name = function() {
return this.firstName + " " + this.lastName;
}
console.log(child.name());

In the prototype object(person), there is a function getAge(). The child object can inherit this function.

If you want a new function to do other things, like print the obejct's name, you can add a new function to the prototype object.


How class allows us to do less typing (syntactic sugar)

class Rectangle {
constructor(height, width) {
this.height = height;
this.width = width;
}
// Getter
get area() {
return this.calcArea();
}
// Method
calcArea() {
return this.height * this.width;
}
}
const square1 = new Rectangle(10, 10);
const square2 = new Rectangle(20,20);
console.log(square1.area); // 100
console.log(square2.area);//400
